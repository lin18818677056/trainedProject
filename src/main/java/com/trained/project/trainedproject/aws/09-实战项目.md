
### 9.4.1 图片上传函数

**文件**: `backend/upload-function/lambda_function.py`

```python
import json
import boto3
import uuid
from datetime import datetime
from PIL import Image
import io

s3 = boto3.client('s3')
dynamodb = boto3.resource('dynamodb')
table = dynamodb.Table('PhotoMetadata')

def lambda_handler(event, context):
    """
    处理图片上传和元数据记录
    """
    try:
        # 获取上传的文件信息
        bucket = event['Records'][0]['s3']['bucket']['name']
        key = event['Records'][0]['s3']['object']['key']
        
        # 生成唯一 ID
        photo_id = str(uuid.uuid4())
        
        # 获取文件元数据
        obj = s3.head_object(Bucket=bucket, Key=key)
        file_size = obj['ContentLength']
        content_type = obj['ContentType']
        
        # 提取图片信息
        img_obj = s3.get_object(Bucket=bucket, Key=key)
        img_data = img_obj['Body'].read()
        img = Image.open(io.BytesIO(img_data))
        width, height = img.size
        
        # 记录到数据库
        item = {
            'PhotoId': photo_id,
            'UserId': 'user-001',  # 实际项目中从认证系统获取
            'S3Key': key,
            'S3Bucket': bucket,
            'Width': width,
            'Height': height,
            'FileSize': file_size,
            'ContentType': content_type,
            'CreatedAt': datetime.now().isoformat(),
            'Tags': []
        }
        
        table.put_item(Item=item)
        
        print(f"Photo uploaded: {photo_id}")
        
        return {
            'statusCode': 200,
            'body': json.dumps({
                'photoId': photo_id,
                'message': 'Upload successful'
            })
        }
        
    except Exception as e:
        print(f"Error: {str(e)}")
        return {
            'statusCode': 500,
            'body': json.dumps({'error': str(e)})
        }
```

### 9.4.2 缩略图生成函数

**文件**: `backend/thumbnail-function/lambda_function.py`

```python
import boto3
from PIL import Image
import io
import os

s3 = boto3.client('s3')

def lambda_handler(event, context):
    """
    自动生成缩略图
    """
    try:
        # 获取触发事件的 S3 信息
        bucket = event['Records'][0]['s3']['bucket']['name']
        key = event['Records'][0]['s3']['object']['key']
        
        # 跳过缩略图文件夹的文件
        if key.startswith('thumbnails/'):
            return {'statusCode': 200, 'body': 'Skipped'}
        
        # 下载原图
        obj = s3.get_object(Bucket=bucket, Key=key)
        image_data = obj['Body'].read()
        
        # 打开并生成缩略图
        img = Image.open(io.BytesIO(image_data))
        
        # 保持宽高比生成缩略图
        img.thumbnail((400, 400), Image.Resampling.LANCZOS)
        
        # 保存到内存
        output = io.BytesIO()
        img.save(output, format=img.format, quality=85)
        output.seek(0)
        
        # 上传缩略图
        thumb_key = f"thumbnails/{key}"
        s3.upload_fileobj(
            output,
            bucket,
            thumb_key,
            ExtraArgs={
                'ContentType': obj['ContentType'],
                'ACL': 'public-read'
            }
        )
        
        print(f"Thumbnail created: {thumb_key}")
        
        return {
            'statusCode': 200,
            'body': 'Thumbnail created successfully'
        }
        
    except Exception as e:
        print(f"Error creating thumbnail: {str(e)}")
        return {
            'statusCode': 500,
            'body': json.dumps({'error': str(e)})
        }
```

### 9.4.3 API 处理函数

**文件**: `backend/api-functions/lambda_function.py`

```python
import json
import boto3
from decimal import Decimal
from datetime import datetime

dynamodb = boto3.resource('dynamodb')
table = dynamodb.Table('PhotoMetadata')

class DecimalEncoder(json.JSONEncoder):
    def default(self, obj):
        if isinstance(obj, Decimal):
            return float(obj)
        return super().default(obj)

def lambda_handler(event, context):
    """
    RESTful API 入口
    """
    http_method = event['httpMethod']
    path = event['path']
    
    # 路由处理
    if http_method == 'GET' and path == '/photos':
        return list_photos(event)
    elif http_method == 'GET' and path.startswith('/photos/'):
        return get_photo(event)
    elif http_method == 'DELETE' and path.startswith('/photos/'):
        return delete_photo(event)
    elif http_method == 'POST' and path == '/photos/search':
        return search_photos(event)
    
    return {
        'statusCode': 404,
        'body': json.dumps({'error': 'Not found'})
    }

def list_photos(event):
    """获取所有照片列表"""
    try:
        response = table.scan()
        photos = response['Items']
        
        # 处理分页（如果有更多数据）
        while 'LastEvaluatedKey' in response:
            response = table.scan(
                ExclusiveStartKey=response['LastEvaluatedKey']
            )
            photos.extend(response['Items'])
        
        return {
            'statusCode': 200,
            'headers': {
                'Content-Type': 'application/json',
                'Access-Control-Allow-Origin': '*'
            },
            'body': json.dumps(photos, cls=DecimalEncoder)
        }
    except Exception as e:
        return {
            'statusCode': 500,
            'body': json.dumps({'error': str(e)})
        }

def get_photo(event):
    """获取单张照片详情"""
    try:
        photo_id = event['pathParameters']['proxy']
        
        response = table.get_item(Key={'PhotoId': photo_id})
        
        if 'Item' in response:
            return {
                'statusCode': 200,
                'headers': {'Access-Control-Allow-Origin': '*'},
                'body': json.dumps(response['Item'], cls=DecimalEncoder)
            }
        else:
            return {
                'statusCode': 404,
                'body': json.dumps({'error': 'Photo not found'})
            }
    except Exception as e:
        return {
            'statusCode': 500,
            'body': json.dumps({'error': str(e)})
        }

def delete_photo(event):
    """删除照片"""
    try:
        photo_id = event['pathParameters']['proxy']
        
        # 先获取照片信息以删除 S3 文件
        response = table.get_item(Key={'PhotoId': photo_id})
        
        if 'Item' not in response:
            return {
                'statusCode': 404,
                'body': json.dumps({'error': 'Photo not found'})
            }
        
        item = response['Item']
        
        # 删除 S3 文件
        s3 = boto3.client('s3')
        s3.delete_object(Bucket=item['S3Bucket'], Key=item['S3Key'])
        s3.delete_object(
            Bucket=item['S3Bucket'],
            Key=f"thumbnails/{item['S3Key']}"
        )
        
        # 删除数据库记录
        table.delete_item(Key={'PhotoId': photo_id})
        
        return {
            'statusCode': 204,
            'headers': {'Access-Control-Allow-Origin': '*'}
        }
    except Exception as e:
        return {
            'statusCode': 500,
            'body': json.dumps({'error': str(e)})
        }

def search_photos(event):
    """搜索照片"""
    try:
        body = json.loads(event['body'])
        user_id = body.get('userId')
        
        if not user_id:
            return {
                'statusCode': 400,
                'body': json.dumps({'error': 'userId required'})
            }
        
        # 使用 GSI 查询
        response = table.query(
            IndexName='UserIdIndex',
            KeyConditionExpression='UserId = :uid',
            ExpressionAttributeValues={':uid': user_id}
        )
        
        photos = response['Items']
        
        return {
            'statusCode': 200,
            'headers': {'Access-Control-Allow-Origin': '*'},
            'body': json.dumps(photos, cls=DecimalEncoder)
        }
    except Exception as e:
        return {
            'statusCode': 500,
            'body': json.dumps({'error': str(e)})
        }
```

### 9.5.3 JavaScript 逻辑

**文件**: `frontend/js/app.js`

```javascript
// API Gateway 端点
const API_ENDPOINT = 'https://xxx.execute-api.us-east-1.amazonaws.com/prod';
const S3_BUCKET = 'photo-app-originals-xxxxxx';
const REGION = 'us-east-1';

// DOM 元素
const uploadArea = document.getElementById('uploadArea');
const fileInput = document.getElementById('fileInput');
const progressBar = document.getElementById('progressBar');
const progress = document.getElementById('progress');
const gallery = document.getElementById('gallery');

// 初始化
document.addEventListener('DOMContentLoaded', () => {
    setupEventListeners();
    loadPhotos();
});

// 设置事件监听
function setupEventListeners() {
    // 点击上传
    uploadArea.addEventListener('click', () => fileInput.click());
    
    // 文件选择
    fileInput.addEventListener('change', handleFiles);
    
    // 拖拽事件
    uploadArea.addEventListener('dragover', (e) => {
        e.preventDefault();
        uploadArea.style.background = '#f8f9ff';
    });
    
    uploadArea.addEventListener('dragleave', () => {
        uploadArea.style.background = '';
    });
    
    uploadArea.addEventListener('drop', (e) => {
        e.preventDefault();
        uploadArea.style.background = '';
        handleDrop(e);
    });
}

// 处理文件上传
async function handleFiles() {
    const files = fileInput.files;
    for (let file of files) {
        await uploadFile(file);
    }
}

// 处理拖拽上传
function handleDrop(e) {
    const files = e.dataTransfer.files;
    for (let file of files) {
        uploadFile(file);
    }
}

// 上传单个文件
async function uploadFile(file) {
    try {
        // 显示进度条
        progressBar.style.display = 'block';
        progress.style.width = '0%';
        
        // 生成唯一文件名
        const timestamp = Date.now();
        const fileName = `${timestamp}-${file.name}`;
        
        // 使用 AWS SDK 直接上传到 S3
        // 这里简化处理，实际应使用预签名 URL
        
        // 模拟上传进度
        let width = 0;
        const interval = setInterval(() => {
            if (width >= 100) {
                clearInterval(interval);
            } else {
                width += 5;
                progress.style.width = width + '%';
            }
        }, 100);
        
        console.log('Uploading:', fileName);
        
        // 上传完成后刷新相册
        setTimeout(() => {
            loadPhotos();
            progressBar.style.display = 'none';
        }, 2000);
        
    } catch (error) {
        console.error('Upload failed:', error);
        alert('上传失败，请重试');
    }
}

// 加载照片列表
async function loadPhotos() {
    try {
        gallery.innerHTML = '<div class="loading">加载中...</div>';
        
        const response = await fetch(`${API_ENDPOINT}/photos`);
        const photos = await response.json();
        
        if (photos.length === 0) {
            gallery.innerHTML = `
                <div class="empty-state">
                    <p>还没有照片，上传第一张吧！</p>
                </div>
            `;
            return;
        }
        
        renderGallery(photos);
        
    } catch (error) {
        console.error('Load failed:', error);
        gallery.innerHTML = '<div class="empty-state">加载失败</div>';
    }
}

// 渲染相册
function renderGallery(photos) {
    gallery.innerHTML = '';
    
    photos.forEach(photo => {
        const card = document.createElement('div');
        card.className = 'photo-card';
        
        const thumbUrl = `https://${photo.S3Bucket}.s3.${REGION}.amazonaws.com/thumbnails/${photo.S3Key}`;
        
        card.innerHTML = `
            <img src="${thumbUrl}" alt="Photo">
            <div class="photo-info">
                <p>尺寸：${photo.Width} x ${photo.Height}</p>
                <p>大小：${(photo.FileSize / 1024).toFixed(1)} KB</p>
                <p>时间：${new Date(photo.CreatedAt).toLocaleDateString()}</p>
                <button class="delete-btn" onclick="deletePhoto('${photo.PhotoId}')">
                    删除
                </button>
            </div>
        `;
        
        gallery.appendChild(card);
    });
}

// 删除照片
async function deletePhoto(photoId) {
    if (!confirm('确定要删除这张照片吗？')) return;
    
    try {
        const response = await fetch(`${API_ENDPOINT}/photos/${photoId}`, {
            method: 'DELETE'
        });
        
        if (response.ok) {
            loadPhotos();
        } else {
            alert('删除失败');
        }
    } catch (error) {
        console.error('Delete failed:', error);
        alert('删除失败');
    }
}
```

## 9.6 部署实施

### 9.6.1 部署 Lambda 函数

**使用 SAM 部署**:

**template.yaml**:
```yaml
AWSTemplateFormatVersion: '2010-09-09'
Transform: AWS::Serverless-2016-10-31

Resources:
  UploadFunction:
    Type: AWS::Serverless::Function
    Properties:
      FunctionName: PhotoUploadHandler
      CodeUri: backend/upload-function/
      Handler: lambda_function.lambda_handler
      Runtime: python3.9
      Timeout: 30
      MemorySize: 512
      Environment:
        Variables:
          DYNAMODB_TABLE: !Ref PhotoTable
      Policies:
        - DynamoDBCrudPolicy:
            TableName: !Ref PhotoTable
        - S3CrudPolicy:
            BucketName: !Ref OriginalBucket
      Events:
        S3Event:
          Type: S3
          Properties:
            Bucket: !Ref OriginalBucket
            Events: s3:ObjectCreated:*
            Filter:
              S3Key:
                Rules:
                  - Name: prefix
                    Value: uploads/

  ThumbnailFunction:
    Type: AWS::Serverless::Function
    Properties:
      FunctionName: ThumbnailGenerator
```
